function generateChecksum(input) {
  let hash = 0;
  for (let i = 0; i < input.length; i++) {
    hash = ((hash << 5) - hash + input.charCodeAt(i)) & 0xffffff;

  return hash.toString(36).substr(-3).toUpperCase();
}
